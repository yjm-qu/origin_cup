## 一、代码实现细节

### 1.1 数据预处理（卷积+池化）

```python
class ConvPreprocessorA(nn.Module):
    def __init__(self):
        self.conv = nn.Sequential(
            nn.Conv2d(1, 4, kernel_size=3, padding=1),  # 16×16×1 → 16×16×4
            nn.ReLU(),
            nn.MaxPool2d(kernel_size=2, stride=2)          # 16×16×4 → 8×8×4
        )
    
    def forward(self, x):
        # 分组池化：每个通道的8×8分成8个条
        for i in range(8):
            col = x[:, :, :, i]      # (batch, 4, 8)
            pooled = col.mean(dim=2) # (batch, 4)
            pooled = pooled.sum(dim=1) # (batch,)
            results.append(pooled)
        return torch.stack(results, dim=1)  # (batch, 8)
```

### 1.2 量子线路

- 量子比特数：8
- Ansatz层数：8
- 每层结构：RY(encoding) → CRZ(theta) → RY(theta)
- 参数数量：120（15 × 8）
- 编码公式：`angle = (π/16) × value`

```python
def build_circuit(encoded_data, params):
    circuit = Circuit(NUM_QUBITS)
    for layer in range(NUM_LAYERS):
        # RY(encoding)
        for i in range(NUM_QUBITS):
            angle = (np.pi / 16) * encoded_data[i]
            circuit.ry(i, angle)
        
        # CRZ(theta)
        for i in range(NUM_CRZ_PER_LAYER):
            theta = params[base_idx + i]
            circuit.crz(i, i + 1, theta)
        
        # RY(theta)
        for i in range(NUM_RY_PER_LAYER):
            theta = params[ry_base_idx + i]
            circuit.ry(i, theta)
    
    circuit.measure(0)
    return circuit
```

### 1.3 损失函数与梯度

```python
def compute_loss(F, target):
    """MSE Loss: L = (y - P)²"""
    return (target - F) ** 2

def compute_gradient(encoded_data, target, params):
    """参数移位法则计算梯度"""
    grad = np.zeros(NUM_PARAMS)
    shift = np.pi / 2
    
    F_current = run_circuit(encoded_data, params)
    coefficient = 2 * (F_current - target)
    
    for i in range(NUM_PARAMS):
        params_plus = params.copy()
        params_plus[i] += shift
        F_plus = run_circuit(encoded_data, params_plus)
        
        params_minus = params.copy()
        params_minus[i] -= shift
        F_minus = run_circuit(encoded_data, params_minus)
        
        grad[i] = coefficient * (F_plus - F_minus) / 2
    
    return grad
```

### 1.4 参数更新（Adam优化器）

```python
LEARNING_RATE = 0.001
BETA1 = 0.9
BETA2 = 0.999
EPSILON = 1e-8

# 参数初始化：均匀分布 (-π, π)
params = np.random.uniform(-np.pi, np.pi, NUM_PARAMS)

# Adam更新
m = BETA1 * m + (1 - BETA1) * grad
v = BETA2 * v + (1 - BETA2) * (grad ** 2)
m_hat = m / (1 - BETA1 ** t)
v_hat = v / (1 - BETA2 ** t)
params = params - LEARNING_RATE * m_hat / (np.sqrt(v_hat) + EPSILON)
```

**BCE Loss 实现**：

```python
def compute_loss(F, target):
    """BCE Loss: -[y*log(P) + (1-y)*log(1-P)]"""
    eps = 1e-8
    P = np.clip(F, eps, 1 - eps)
    if target == 1:
        return -np.log(P)
    else:
        return -np.log(1 - P)

def compute_gradient(encoded_data, target, params):
    """BCE Loss 梯度"""
    F_current = run_circuit(encoded_data, params)
    eps = 1e-8
    P = np.clip(F_current, eps, 1 - eps)
    
    if target == 1:
        dL_dP = -1.0 / P
    else:
        dL_dP = 1.0 / (1 - P)
    
    for i in range(NUM_PARAMS):
        # ... 计算 F_plus, F_minus
        dP_dtheta = (F_plus - F_minus) / 2
        grad[i] = dL_dP * dP_dtheta
    
    return grad
```

## 四、问题分析

### 4.1 损失值分析

方案一的损失降到约0.25后难以继续下降。

当 MSE Loss ≈ 0.25 时：
- (y - P)² = 0.25
- |y - P| = 0.5
- 当 y=0 时，P=0.5
- 当 y=1 时，P=0.5

**问题**：模型输出趋向于0.5，无法有效区分两类样本。

### 4.2 可能原因

1. **量子电路表达能力有限**：8量子比特、8层可能不足以区分数字4和9
2. **测量方式单一**：仅测量1个量子比特的信息
3. **编码方式可能不合适**：`angle = (π/16) × value` 的范围可能需要调整
4. **数据预处理**：卷积+池化可能丢失了关键特征
